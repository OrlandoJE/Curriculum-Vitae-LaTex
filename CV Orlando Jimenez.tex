\newbool{showprogramming}

% Set the booleans based on the selected profile
\ifx\cvprofile\undefined
  \booltrue{shownano}
  \booltrue{showprogramming}
\else
  \ifnum\pdfstrcmp{\cvprofile}{nano}=0
    \booltrue{shownano}
    \boolfalse{showprogramming}
  \else
    \ifnum\pdfstrcmp{\cvprofile}{programming}=0
      \boolfalse{shownano}
      \booltrue{showprogramming}
    \else
      \ifnum\pdfstrcmp{\cvprofile}{both}=0
        \booltrue{shownano}
        \booltrue{showprogramming}
      \else
        \booltrue{shownano}
        \booltrue{showprogramming}
      \fi
    \fi
  \fi
\fi
